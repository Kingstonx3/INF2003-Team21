# Read Mongo database
from pymongo import MongoClient

mongo_uri = 'mongodb://localhost:27017/'
client = MongoClient(mongo_uri)
db = client['databaseProject']
collection = db['adminPriv']

# Retrieve all documents from the collection
all_entries = collection.find()

# Print all entries
for entry in all_entries:
    print(entry)

# Close the MongoDB connection
client.close()



update_values = {
        "userId": 2,
        "moviePerm": True,
        "accPerm": True
}

# Update the document that matches the conditions or insert a new one if no match is found
result = collection.insert_one(update_values)
